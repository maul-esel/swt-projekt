title Seite einsprechen

":Redakteur" ->+ ":System": getWorkbooks()
":System" -->- ":Redakteur": workbooks

opt Arbeitsheft existiert nicht
  ":Redakteur" ->+ ":System": createWorkbook(title, subtitle)
  ":System" -->- ":Redakteur": workbook
end

":Redakteur" ->+ ":System": createNewPage(workbook, pageNr)
":System" -->- ":Redakteur": page

loop
  alt Neue Aufnahme
    ":Redakteur" ->+ ":System": beginRecording()
    ":System" -->- ":Redakteur": recording
    ":Redakteur" -> ":System": endRecording(recording)
  else Aufnahme importieren
    ":Redakteur" ->+ ":System": importRecording(fileName)
    ":System" -->- ":Redakteur": recording
  end
end

":Redakteur" -> ":System": savePage(page, recording)
